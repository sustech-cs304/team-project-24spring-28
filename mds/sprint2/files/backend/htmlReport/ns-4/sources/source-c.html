     private Event toEvent;

     private Post toPost;

     private Comment toComment;
 
     @ManyToOne(fetch = FetchType.LAZY)

     private boolean read;
 
     private LocalDateTime time;
 
     @Column(columnDefinition = "text")
     private String content;
